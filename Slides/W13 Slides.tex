\begin{frame}{Monopolistic Competition}
    \begin{block}{What is monopolistic competition?}
    Many firms sell products that are similar but not identical
        \begin{itemize}
            \item Many buyers and sellers
            \item Products are \textit{similar but not identical}
            \item Free entry and exit (in the long run)
        \end{itemize}
    \end{block}
    \begin{center}
        What are some examples?
    \end{center}
\end{frame}

\begin{frame}{Monopolistic Competition}
    Just like any other type of market...
    \begin{center}
        \textbf{Firms maximize profit when they set $MR = MC$!}
    \end{center}
    \begin{block}{What will the firms demand curve look like?}
    \pause
        \begin{itemize}
            \item Firms are not price takers - they have some market power due to differentiated products
            \item Firm demand will resemble that of a monopoly
            \item It is downward sloping!
        \end{itemize}
    \end{block}
    \vspace{4mm}
\begin{center}
    Let's draw this market...
\end{center}
\end{frame}

\begin{frame}{Short vs. Long run}
    \begin{block}{In the short-run:}
        \begin{itemize}
            \item If $P > ATC \implies$ positive $\pi$
            \item If $P < ATC \implies$ loss
As long as $P > AVC$, the firm continues to operate
        \end{itemize}
    \end{block}

    \begin{block}{In the long-run:}
        \begin{itemize}
            \item When firms make losses in the SR $\implies$ exit
            \begin{itemize}
                \item $\to$ increases demand for each existing firm
                \item $\to$ $\uparrow P \quad \& \; \uparrow \pi$
            \end{itemize}
            \item When firms make positive $\pi \implies$ new firms enter \& produce close substitutes
            \begin{itemize}
                \item $\to$ reduces demand for each existing firm
                \item $\to$ $\downarrow P \quad\& \; \downarrow \pi$
            \end{itemize}
        \end{itemize}
\begin{center}
\textbf{** entry/exit $\implies$ all firms earn 0 economic $\pi$ in LR!! **}
\end{center}
    \end{block}
\end{frame}

\begin{frame}{In the long run...}
Two important characteristics of monopolistic competition:
    \begin{enumerate}
        \item As in a monopoly market: $MC < P$ - \textbf{Markup}
        \begin{itemize}
            \item maximizing $\pi$ means setting $MR = MC$, and $MR < $ demand curve, which gives us price
            \item So $MR < P$
        \end{itemize}
    \item As in a competitive market: $P = ATC$ - \textbf{No profits}
    \begin{itemize}
        \item free entry and exit drive economic profit to 0
    \end{itemize}
    \end{enumerate}
\vspace{3mm}
    The optimal $Q$ for a monopolistically competitive firm is $Q$ such that:
    \begin{enumerate}
        \item $MR = MC$
        \item $ATC$ is tangent to (kisses) the firms demand curve
    \end{enumerate}
\end{frame}

\begin{frame}{Monopolistic vs. Perfect Competition}
    Two key differences:
    \begin{block}{Excess capacity:}
        The \textbf{efficient scale} is the $Q$ which minimizes $ATC$
        \begin{itemize}
            \item A monopolistically competitive firm has excess capacity:
            \begin{itemize}
                \item They could $\uparrow Q$ and that would $\downarrow ATC$
            \end{itemize}
        \end{itemize}
    \end{block}

    \begin{block}{Markup:}
        A monopolistically competitive firm operates where:
        \begin{itemize}
            \item $ATC$ is decreasing $\to$ where $MC < ATC$
            \item $\implies MC < P$
            \item firms want to induce additional demand
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Monopolistic competition and welfare}
\textbf{Externality:} uncompensated impact of one person/firms actions on a bystander person/firm
\begin{itemize}
    \item Positive externality: surplus of bystander goes up
    \item Negative externality: surplus of bystander goes down
\end{itemize}

\begin{block}{How does firm entry affect a monopolistically competitive market?}
    \begin{itemize}
    \item \textbf{Product-variety externality:} positive externality on consumers
    \item \textbf{Business-stealing externality:} negative externality on sellers
    \end{itemize}
\end{block}
\end{frame}

\begin{frame}{What is an Oligopoly?}
    \begin{block}{An \textbf{oligopoly} is a market in which:}
        \begin{itemize}
            \item there are only a few sellers
            \item they offer similar or identical products
        \end{itemize}
    \end{block}
\vspace{2mm}
    A \textbf{Duopoly} is an oligopoly with only 2 sellers
    \begin{itemize}
        \item We usually consider duopolies since they have the same outcomes/properties as larger oligopolies
    \end{itemize}

\begin{frame}{How does an Oligopoly determine $P^*$ \& $Q^*$?}
    When there are a small number of firms, as in an oligopoly, those firms can collude...
    \begin{block}{\textbf{Collusion} is when:}
       oligopoly firms make an agreement over prices and production
    \end{block}
    \begin{block}{A \textbf{cartel} is:}
        A group of oligopoly firms which are colluding together
    \end{block}
\end{frame}

\begin{frame}{Nash Equilibrium}
    The interaction between Verizon and Xfinity tells us something about the strategic behavior of oligopolies...

    \begin{block}{A nash equilibrium is:}
        A set of strategies such that economic actors interacting together each choose their best strategy given the strategies that the others have chosen
    \end{block}
    \vspace{4mm}
    When Verizon and Xfinity act in their own self interest by increasing production, they lose out on profit.\\
    $\to$ So they would rather collude to produce monopoly $P$ \& $Q$
    \begin{center}
        \textit{But will they cooperate? Is this easy?}
    \end{center}
\end{frame}

\begin{frame}{The Prisoner's Dilemma}
    Let's consider a "Game" called the Prisoner's Dilemma:
    \begin{itemize}
        \item 2 criminals (Bonnie and Clyde) stole \$100 and were caught
        \item Each has the option: Confess or Stay silent
        \item There are 3 possible outcomes
        \begin{itemize}
            \item Both stay silent = they keep \$100 and split it 50-50
            \item One confesses and the other remains silent = silent one gets nothing while confessor gets \$60
            \item Both confess: they split the \$60 reward 50-50
        \end{itemize}
    \end{itemize}
\begin{center}
    What does this look like? How do we find the nash equilibrium?
\end{center}
\end{frame}

\begin{frame}{Cooperation}
    \begin{itemize}
        \item Cooperation is hard to maintain!  \quad Even when it would make you better off
        \item The same principles apply to oligopolies and their choice of production
        \begin{itemize}
            \item collusion at the monopoly $P$ \& $Q$ is rational for both providers
            \item BUT each firm has an incentive to cheat
            \item and each firm anticipates the other cheating
        \end{itemize}
    \end{itemize}



\vspace{1cm}

\begin{frame}{Externaltities}
   
   Recall... \textbf{Externality:} uncompensated impact of one person/firms actions on a bystander person/firm
\begin{itemize}
    \item Positive externality: surplus of bystander goes up
    \item Negative externality: surplus of bystander goes down
\end{itemize}
\vspace{2mm}
We can \textbf{internalize an externality}\\
\quad $\to$ alter incentives so that people take the external impacts of their actions into account

\begin{itemize}
    \item Externalities are often cited as reasons for government intervention:
    \begin{itemize}
        \item \textbf{Corrective/Pigouvian tax:} a tax designed to alter buyer demand through changes in price to bring the market equilibrium quantity closer to the social optimal level
        \item Forces individuals to internalize the external impacts of their actions
        \item Examples?
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Internalizing an externality}
\begin{block}{Consider the market for cigarettes}
    \begin{itemize}
        \item Second hand smoke is bad for bystanders $\to$ cigarettes cause a negative externality
        \item So the socially optimal quantity $Q^{SO} < Q^*$, the equilibrium quantity 
    \end{itemize}
    \begin{center}
        \textit{The government can tax cigarettes to reduce the equilibrium quantity to the socially optimal level}
    \end{center}
\end{block}

\end{frame}
